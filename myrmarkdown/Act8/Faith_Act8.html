7. Create a for loop function to explore Ks from 2 to 10 for each method using the Silhouette score and display the results with a plot.

wardmorph_avg <- numeric()

for (i in 2:10){
  wardmorph_clust <- cutree(wardmorph, i)
  wardmorph_sil <- silhouette(wardmorph_clust, euc)
  wardmorph_avg[i] <- mean(wardmorph_sil[, 3])}

plot(
  2:10,
  wardmorph_avg[2:10],
  xlab = "# of Clusters",
  ylab = "Silhoutte Score",
  main = "Silhoutte Score for Single Linkage",
  lwd = 2,
  cex = 1.5,
